## 🔧 技术实现细节

### 1. 动态 Meta 标签设置

**HomePage.tsx** 示例：
```tsx
useEffect(() => {
  document.title = `${BLOG_NAME} - ${BLOG_DESC}`;
  
  const setMeta = (name: string, content: string, type = 'name') => {
    let meta = document.querySelector(`meta[${type}="${name}"]`);
    if (!meta) {
      meta = document.createElement('meta');
      meta.setAttribute(type, name);
      document.head.appendChild(meta);
    }
    meta.setAttribute('content', content);
  };
  
  setMeta('description', BLOG_DESC);
  setMeta('author', 'AI进化论');
  setMeta('og:title', `${BLOG_NAME} - ${BLOG_DESC}`);
  setMeta('og:description', BLOG_DESC);
  setMeta('og:url', BLOG_URL);
  setMeta('og:type', 'website');
  setMeta('og:site_name', BLOG_NAME);
  setMeta('twitter:card', 'summary_large_image');
  setMeta('twitter:title', `${BLOG_NAME} - ${BLOG_DESC}`);
  setMeta('twitter:description', BLOG_DESC);
}, []);
```

### 2. Canonical URL 设置函数

```tsx
// src/services/seoService.ts
export const setCanonicalUrl = (url: string) => {
  let link: HTMLLinkElement = document.querySelector('link[rel="canonical"]');
  if (!link) {
    link = document.createElement('link');
    link.rel = 'canonical';
    document.head.appendChild(link);
  }
  link.href = url;
};
```

### 3. JSON-LD 注入函数

```tsx
export const injectStructuredData = (schema: object, id: string) => {
  const existing = document.getElementById(id);
  if (existing) existing.remove();
  
  const script = document.createElement('script');
  script.type = 'application/ld+json';
  script.id = id;
  script.textContent = JSON.stringify(schema);
  document.head.appendChild(script);
};
```
